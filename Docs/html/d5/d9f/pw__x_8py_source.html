536def build_PW_Input(scf_inputs, material_prefix):
537 print('BUILDING PW INPUT FILE')
538 #Build pw input file
539 scf_name = 'scf-%s.in' % material_prefix #assigns pw input file to variable pw_name

541 if ibrav.value == 0:

580 &control
581 calculation = 'scf'
582 prefix = '{material_prefix}'
583
584 restart_mode = '{restart_mode}'
