640 outnscf = f'{material_prefix.value}nscf'
641 with open(f'{pwd}/{outnscf}.stdout') as file:
642 for line in file:
643 j = line.split()
644 if 'highest' in j and 'lowest' in j:
645 fermi_energy = (float(j[len(j)-1])+float(j[len(j)-2]))/2
